20170505-函数作用域与this指向

1，作用域

变量间关系

    var num=10;
    var num1=new Number(10);
    var num2=new Number(10);
    var num3=num2;
    num.k=8;
    num1.kkk=8;
    alert(num==num1);//true
    alert(num1==num2);//false,两个对象比较
    alert(num2==num3);//true,比较两个地址
    alert(num.constructor);//function Number(){[native code]},num隐式转换成number对象，但num本身还是数字
    alert(num.k);//undefined,本身还是数字，不能创建属性k
    alert(num1.kkk);//num1是number对象，创建属性kkk

2,this的指向

call函数和apply方法

call函数和apply方法的第一个参数都是要传入给当前对象的对象，即函数内部的this。后面的参数都是传递给当前对象的参数。

    var func=new function(){this.a="func"}
    var myfunc=function(x){
       var a="myfunc";
       alert(this.a);       //func
       alert(x);            //var
    }
    myfunc.call(func,"var");
先调用func函数，用this.a=”func”替换myfunc中的this.a ; 然后将”var”传递给方法myfunc的参数x。
    function Obj(){this.value="对象！";}
    var value="global 变量";
    function Fun1(){alert(this.value);}
    window.Fun1();   //global 变量
    Fun1.call(window);  //global 变量
    Fun1.call(document.getElementById('myText'));  //input text
    Fun1.call(new Obj());   //对象！
    window.Fun1(); //global 变量
